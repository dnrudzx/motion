101  std::pair<bool, std::shared_ptr<std::vector<std::shared_ptr<TDatum>>>> DatumProducer<TDatum>::checkIfRunningAndGetDatum()
102  {
103  try
104  {
105  auto datums = std::make_shared<std::vector<std::shared_ptr<TDatum>>>();
106  // Check last desired frame has not been reached
107  if (mNumberFramesToProcess != std::numeric_limits<unsigned long long>::max()
108  && mGlobalCounter > mNumberFramesToProcess)
109  {
110  spProducer->release();
111  }
112  // If producer released -> it sends an empty cv::Mat + a datumProducerRunning signal
113  const bool datumProducerRunning = spProducer->isOpened();
114  // If device is open
115  if (datumProducerRunning)
116  {
117  // Fast forward/backward - Seek to specific frame index desired
118  if (spVideoSeek != nullptr)
119  {
120  // Fake pause vs. normal mode
121  const auto increment = spVideoSeek->second - (spVideoSeek->first ? 1 : 0);
122  // Normal mode
123  if (increment != 0)
124  spProducer->set(CV_CAP_PROP_POS_FRAMES, spProducer->get(CV_CAP_PROP_POS_FRAMES) + increment);
125  // It must be always reset or bug in fake pause
126  spVideoSeek->second = 0;
127  }
128  auto nextFrameName = spProducer->getNextFrameName();
129  const auto nextFrameNumber = (unsigned long long)spProducer->get(CV_CAP_PROP_POS_FRAMES);
130  const auto cvMats = spProducer->getFrames();
131  const auto cameraMatrices = spProducer->getCameraMatrices();
132  auto cameraExtrinsics = spProducer->getCameraExtrinsics();
133  auto cameraIntrinsics = spProducer->getCameraIntrinsics();
134  // Check frames are not empty
135  checkIfTooManyConsecutiveEmptyFrames(mNumberConsecutiveEmptyFrames, cvMats.empty() || cvMats[0].empty());
136  if (!cvMats.empty())
137  {
138  datums->resize(cvMats.size());
139  // Datum cannot be assigned before resize()
140  auto& datumPtr = (*datums)[0];
141  datumPtr = std::make_shared<TDatum>();
142  // Filling first element
143  std::swap(datumPtr->name, nextFrameName);
144  datumPtr->frameNumber = nextFrameNumber;
145  datumPtr->cvInputData = cvMats[0];
146  if (!cameraMatrices.empty())
147  {
148  datumPtr->cameraMatrix = cameraMatrices[0];
149  datumPtr->cameraExtrinsics = cameraExtrinsics[0];
150  datumPtr->cameraIntrinsics = cameraIntrinsics[0];
151  }
152  // Image integrity
153  if (datumPtr->cvInputData.channels() != 3)
154  {
155  const std::string commonMessage{"Input images must be 3-channel BGR."};
156  // Grey to RGB if required
157  if (datumPtr->cvInputData.channels() == 1)
158  {
159  log(commonMessage + " Converting grey image into BGR.", Priority::High);
160  cv::cvtColor(datumPtr->cvInputData, datumPtr->cvInputData, CV_GRAY2BGR);
161  }
162  else
163  error(commonMessage, __LINE__, __FUNCTION__, __FILE__);
164  }
165  datumPtr->cvOutputData = datumPtr->cvInputData;
166  // Resize if it's stereo-system
167  if (datums->size() > 1)
168  {
169  // Stereo-system: Assign all cv::Mat
170  for (auto i = 1u ; i < datums->size() ; i++)
171  {
172  auto& datumIPtr = (*datums)[i];
173  datumIPtr = std::make_shared<TDatum>();
174  datumIPtr->name = datumPtr->name;
175  datumIPtr->frameNumber = datumPtr->frameNumber;
176  datumIPtr->cvInputData = cvMats[i];
177  datumIPtr->cvOutputData = datumIPtr->cvInputData;
178  if (cameraMatrices.size() > i)
179  {
180  datumIPtr->cameraMatrix = cameraMatrices[i];
181  datumIPtr->cameraExtrinsics = cameraExtrinsics[i];
182  datumIPtr->cameraIntrinsics = cameraIntrinsics[i];
183  }
184  }
185  }
186  // Check producer is running
187  if (!datumProducerRunning || (*datums)[0]->cvInputData.empty())
188  datums = nullptr;
189  // Increase counter if successful image
190  if (datums != nullptr)
191  mGlobalCounter += mFrameStep;
192  }
193  }
194  // Return result
195  return std::make_pair(datumProducerRunning, datums);
196  }
197  catch (const std::exception& e)
198  {
199  error(e.what(), __LINE__, __FUNCTION__, __FILE__);
200  return std::make_pair(false, std::make_shared<std::vector<std::shared_ptr<TDatum>>>());
201  }
